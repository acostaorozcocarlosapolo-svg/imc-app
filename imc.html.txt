<script>
function calcular() {
    let peso = parseFloat(document.getElementById("peso").value);
    let altura = parseFloat(document.getElementById("altura").value);

    if (!peso || !altura) {
        document.getElementById("resultado").innerHTML = 
            "Por favor, ingresa valores válidos.";
        return;
    }

    let imc = peso / (altura * altura);
    let categoria = "";
    let consejo = "";

    if (imc < 18.5) {
        categoria = "Bajo peso";
        consejo = "Aumenta calorías saludables y entrena fuerza para ganar masa muscular.";
    } else if (imc < 24.9) {
        categoria = "Normal";
        consejo = "Mantén hábitos saludables, buena hidratación y ejercicio semanal.";
    } else if (imc < 29.9) {
        categoria = "Sobrepeso";
        consejo = "Reduce azúcares y comida procesada, aumenta actividad física.";
    } else {
        categoria = "Obesidad";
        consejo = "Consulta un profesional y comienza actividad física gradual.";
    }

    document.getElementById("resultado").innerHTML = `
        <div class="categoria">${categoria}</div>
        Tu IMC es: <strong>${imc.toFixed(2)}</strong><br><br>
        <em>${consejo}</em>
    `;
}
</script>
